213bool Calibrate_Acc(){
214 /* Loop para leer los valores iniciales */
215 x_promedio = 0;
216 y_promedio = 0;
217 z_promedio = 0;
218 for (uint16_t i = 0; i < LECTURAS_CALIBRACION; i++)
219 {
220 /* Sumatoria de los primeros *LECTURAS_CALIBRACION* valores */
221 x_promedio = x_promedio + ReadXValue();
222 y_promedio = y_promedio + ReadYValue();
223 z_promedio = z_promedio + ReadZValue();
224 }
225
226 /* Calculo de los promedios */
227 x_promedio = x_promedio/LECTURAS_CALIBRACION;
228 y_promedio = y_promedio/LECTURAS_CALIBRACION;
229 z_promedio = z_promedio/LECTURAS_CALIBRACION;
230
231 return 1;
232}
